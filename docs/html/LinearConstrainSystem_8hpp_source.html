109void LinearConstrainSystem<T>::update_tableau_info() {
110
111 // updating number of constrains in tableau
112 tab.num_constrains = constrains.size();
113 // updating number of decisional variables in tableau
114 tab.num_variables = constrains[0].a.size();
115 // updating number of added variables in tableau (i.e. slack, surplus, artificial)
116 for (const auto& constrain : constrains) {
117

172template <typename T>
173bool LinearConstrainSystem<T>::is_feasible() {
174
175 // checking validity of input values
176 check_valid_constrains();
177 // updating input information received so far
178 update_tableau_info();
179 // creating a copy of the linear constrain system received so far
180 LinearConstrainSystem<T> copy(*this);
181
182 // updating dimension of constrains inserted so far to make place for the dummy variable
183 for (size_t i=0; i< copy.constrains.size(); ++i) {
184 copy.constrains[i].a.push_back(0);
185 }
186 // creating constrain coefficients referring to dummy variable
187 std::vector<T> a_dummy(copy.constrains[0].a.size()-1, 0);
188 a_dummy.emplace_back(1);
189 // adding constrain to constrain system
190 copy.add_constrain(a_dummy, 0 ,ConstrainType::EQ);
191 // updating dummy variable information inside tableau
192 copy.tab.num_constrains++;
193 copy.tab.num_variables++;
194 copy.tab.artificial_variables++;
195 // creating initial tableau
196 copy.tab.create_initial_tableau(copy.constrains);
197 // creating dummy objective variable
198 std::vector<T> c(copy.tab.num_variables - 1, 0);
199 c.emplace_back(1);
200 // adding dummy objective variable to tableau
201 copy.tab.add_objFunc_tableau(c, LinearConstrainSystem<T>::OptimizationType::MAX);
202
203 // executing symplex pivot method until it gets interrupted
204 bool hasSimplexFinished = false;
205 while (!hasSimplexFinished ) {
206 // obtaining index of base-entering variable
207 int pivot_column = copy.tab.find_pivot_column();
208 // if the index is -1 then there are not variables to enter the base anymore: the loop interrupts
209 if (pivot_column == -1 ) {
210 #ifdef PRINT
211 std::cout << "----End Simplex----" << std::endl<<std::endl;
212 #endif // PRINT
213 hasSimplexFinished = true;
214 }
215 if (hasSimplexFinished == false) {
216 // obtaining index of base-exiting variable
217 int pivot_row = copy.tab.find_pivot_row(pivot_column);
218 // pivot executes
219 copy.tab.pivot(pivot_row, pivot_column);
220 }
221 }
222 #ifdef PRINT
223 std::cout<< "FEASIBILITY TEST: "<< std::endl<<std::endl;
224 #endif // PRINT
225
226 T solution = copy.tab.tableau[copy.tab.num_constrains].back();
227 // if z<0 then the system is infeasible
228 if (solution < 0) {
229 throw std::runtime_error("The linear constraint system is INFEASIBLE.");
230 return false;
231 // otherwise it is feasible
232 } else {
233 #ifdef PRINT
234 std::cout<< "The system is FEASIBLE!"<< std::endl<<std::endl;
235 #endif // PRINT
236 // updating the flag of feasibility_test
237 feasibility_test = true;
238 return true;
239 }

252template<typename T>
253typename LinearConstrainSystem<T>::SolutionType LinearConstrainSystem<T>::optimize(std::vector<T>& solution,
254 const std::vector<T>& c,
255 const OptimizationType type) {
256
257 // variable that will be returned
258 LinearConstrainSystem<T>::SolutionType sol_type;
259 // if user has not executed is_feasible then do it
260 if (feasibility_test == false) {
261 is_feasible();
262 }
263 // checking input objective function
264 check_valid_objFunc(c, type);
265 // creating a copy of the LinearConstrainSystem obtained so far
266 LinearConstrainSystem<T> copy(*this);
267 // creating initial tableau
268 copy.tab.create_initial_tableau(copy.constrains);
269 // adding objective function row to tableau
270 copy.tab.add_objFunc_tableau(c, type);
271
272 // SIMPLEX ALGORITHM PROCEDURE:
273 // Executing pivot method until it gets interrupted
274 bool hasSimplexFinished = false;
275 while (!hasSimplexFinished) {
276
277 // obtaining base-entering variable index
278 int pivot_column = copy.tab.find_pivot_column();
279 // if pivot column is -1 then there are no variable that can be set in base anymore; symplex is interrupted
280 if (pivot_column == -1 ) {
281 #ifdef PRINT
282 std::cout << "----End Simplex----" << std::endl<<std::endl;
283 #endif // PRINT
284 hasSimplexFinished = true;
285 }
286 if (hasSimplexFinished == false) {
287 // obtaining base-exiting variable index
288 int pivot_row = copy.tab.find_pivot_row(pivot_column);
289 // if pivot row is -1 then the system is unbounded
290 if (pivot_row == -1) {
291 sol_type = SolutionType::UNBOUNDED; // update and then return the variable
292 print_Lcs(c,type);
293 print_result(sol_type, solution);
294 return sol_type;
295 }
296 // performing pivot method
297 copy.tab.pivot(pivot_row, pivot_column);
298 }
299 }
300 // writing found solution
301 solution.resize(copy.tab.num_variables);
302
303 for (size_t i = 0; i < copy.tab.num_variables; ++i) {
304 // getting indexes of decisional variables
305 size_t decision_variable = copy.tab.get_decVars_index() + i;
306 // searching base vector for indexes of decisional variables
307 auto index = std::find(copy.tab.base.begin(), copy.tab.base.end(), decision_variable);
308
309 if (index != copy.tab.base.end()) {
310 // tableau row index corresponding to base variable found
311 size_t row = index - copy.tab.base.begin();
312 // taking last value of the row (i.e. the known term) and saving it in solution
313 solution[i] = copy.tab.tableau[row].back();
314 }
315 }
316 // saving z value at the end of solution vector
317 solution.emplace_back(copy.tab.tableau[copy.tab.num_constrains].back());
318 // printing optimization problem
319 sol_type = SolutionType::BOUNDED;
320 print_Lcs(c,type);
321 print_result(sol_type, solution);
322
323 return sol_type;
324}
